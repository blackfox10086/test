pipeline {
    agent any
    stages {
        stage('SAST') {
            steps {
                sh 'semgrep ci --config=p/r2c || true'
            }
        }
        stage('SCA') {
            steps {
                sh 'npm ci && npm audit --audit-level high --json > npm-audit.json || true'
            }
        }
        stage('DAST') {
            steps {
                sh 'zap-baseline.py -t http://localhost:3030 -r zap-report.html || true'
            }
        }
        stage('Deploy 3030') {
            steps {
                sh '''
                docker stop donation-app || true
                docker rm donation-app || true
                docker build -t donation-app .
                docker run -d -p 3030:3000 --name donation-app donation-app
                '''
            }
        }
    }
    post {
        always {
            publishHTML([allowMissing: false, alwaysLinkToLastBuild: true, keepAll: true, reportDir: '.', reportFiles: 'npm-audit.json,zap-report.html', reportName: 'Security Reports'])
            archiveArtifacts artifacts: '.json,.html', allowEmptyArchive: true
        }
    }
}
